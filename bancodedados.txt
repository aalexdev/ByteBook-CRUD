CREATE DATABASE ByteBook;
USE ByteBook;

-- 1. TABELA: Clientes
CREATE TABLE clientes (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    telefone VARCHAR(15) NULL,
    cpf VARCHAR(14) NOT NULL UNIQUE
);

-- 2. TABELA: Livros
CREATE TABLE livros (
    id INT IDENTITY(1,1) PRIMARY KEY,
    titulo VARCHAR(100) NOT NULL,
    autor VARCHAR(100) NOT NULL,
    isbn VARCHAR(20) NULL,
    genero VARCHAR(50) NULL,
    ano INT NULL,
    quantidade INT NOT NULL DEFAULT 1
);

-- 3. TABELA: Empréstimos (Integrado direto com Clientes e Livros para o Frontend)
CREATE TABLE emprestimos (
    id INT IDENTITY(1,1) PRIMARY KEY,
    livro_id INT NOT NULL FOREIGN KEY REFERENCES livros(id) ON DELETE CASCADE,
    cliente_id INT NOT NULL FOREIGN KEY REFERENCES clientes(id) ON DELETE CASCADE,
    data_emprestimo DATE NOT NULL DEFAULT GETDATE(),
    data_devolucao DATE NULL,
    devolvido BIT NOT NULL DEFAULT 0 -- 0 = Ativo, 1 = Devolvido
);
